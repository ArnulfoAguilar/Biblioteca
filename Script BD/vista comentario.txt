CREATE OR REPLACE VIEW opcionescomentario AS
SELECT c.id AS "ID_COMENTARIO",
    c."COMENTARIO",
    c."HABILITADO",
    c."ID_USUARIO" AS "AUTOR_COMENTARIO",
    c."ID_APORTE",
    c.created_at AS "DATE_CREA_COMENT",
    c.updated_at AS "DATE_UPD_COMENT",
    ic."ID_TIPO_INTERACCION",
    ic."ID_COMENTARIO" AS "ID_COMENT_FK",
    ic."ID_USUARIO" AS "ID_USUARIO_INTERACCION",
    ic.created_at AS "DATE_CREATE_LIKE",
    ic.updated_at AS "DATE_UPD_LIKE",
		( SELECT users_1.name AS autor_co
           FROM users users_1
          WHERE (ic."ID_USUARIO" = users_1.id)) AS autor_co
   FROM (("Comentario" c
     left JOIN "interaccionComentario" ic ON ((c.id = ic."ID_COMENTARIO")))
     JOIN users ON ((c."ID_USUARIO" = users.id)))
