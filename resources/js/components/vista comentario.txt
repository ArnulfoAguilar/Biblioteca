 SELECT c.id AS "ID_COMENTARIO",
    c."COMENTARIO",
    c."HABILITADO",
    c."ID_USUARIO" AS "AUTOR_COMENTARIO",
    c."ID_APORTE",
    c.created_at AS "DATE_CREA_COMENT",
    c.updated_at AS "DATE_UPD_COMENT",
    bc."LIKE",
    bc."ID_COMENTARIO" AS "ID_COMENT_FK",
    bc."ID_USUARIO" AS "ID_USUARIO_LIKE",
    bc.created_at AS "DATE_CREATE_LIKE",
    bc.updated_at AS "DATE_UPD_LIKE",
		( SELECT users_1.name AS autor_co
           FROM users users_1
          WHERE (bc."ID_USUARIO" = users_1.id)) AS autor_co
   FROM (("Comentario" c
     left JOIN "bitacoraComentario" bc ON ((c.id = bc."ID_COMENTARIO")))
     JOIN users ON ((c."ID_USUARIO" = users.id)))
